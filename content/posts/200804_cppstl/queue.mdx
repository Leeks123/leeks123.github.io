###### 큐

```c++
#include <iostream>
#include <queue>
using namespace std;
int main(){
	
	queue<int> q;//int자료형을 저장하는 큐 생성
	
	q.push(4); // 맨 뒤에 요소 삽입
	q.pop(); // 맨 앞의 요소를 반환
	
  q.front();//큐의 맨 앞에 있는 요소에 대한 참조를 반환
  q.back();//큐의 맨 뒤에 있는 요소에 대한 참조를 반환
  
  q.empty(); //큐가 비어있으면 true, 비어있지 않으면 false 반환
	
  q.size(); //큐에 저장되어 있는 원소의 개수
  
	return 0;
}
```
